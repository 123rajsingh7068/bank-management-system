BankManagement

create database banksystem;
use banksystem;

create table signup(
FormNo varchar(100),
Name varchar(100),
FatherName varchar(100),
Gender varchar(100),
BOB varchar(100),
MartialStatus varchar(100),
Email varchar(100),
Address varchar(100),
City varchar(100),
PinCode varchar(100),
State varchar(100)
);
select * from signup;
truncate table signup;

create table signuptwo(
FormNo varchar(100),
Religion varchar(100),
Category varchar(100),
Income varchar(100),
Education varchar(100),
Occupation varchar(100),
Pan varchar(100),
Aadhar varchar(100),
SeniorCitizen varchar(100),
ExistingAccount varchar(100)
);
select * from signuptwo;
truncate table signuptwo;

create table signupthree(
FormNo varchar(100),
AccountType varchar(100),
CardNo varchar(100),
Pin varchar(100),
Facility varchar(100)
);
select * from signupthree;
truncate table signupthree;

create table Login(
FormNo varchar(100),
CardNo varchar(100),
Pin varchar(100)
);
select * from Login;
truncate table Login;

create table Deposit (
Pin varchar(100),
Data varchar(100),
Type varchar(100),
Amount varchar(100)
);
select * from Deposit;
truncate table Deposit;
